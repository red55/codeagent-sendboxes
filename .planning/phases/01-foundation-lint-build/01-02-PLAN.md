Create `.github/workflows/pr.yml` with the following structure:

```yaml
name: PR Build Validation

on:
  pull_request:
    branches: [main]

permissions:
  contents: read
  security-events: write

concurrency:
  group: ${{ github.workflow }}-${{ github.ref }}
  cancel-in-progress: true

jobs:
  lint:
    name: Lint
    runs-on: ubuntu-latest
    steps:
      - name: Checkout
        uses: actions/checkout@v6

      - name: Run hadolint
        id: hadolint
        uses: hadolint/hadolint-action@v3.1.0
        with:
          dockerfile: base/Dockerfile
          config: .github/lint/.hadolint.yaml
          verbose: true

      - name: Run hadolint on base/opencode.Dockerfile
        uses: hadolint/hadolint-action@v3.1.0
        with:
          dockerfile: base/opencode.Dockerfile
          config: .github/lint/.hadolint.yaml
          verbose: true

      - name: Run hadolint on base/qwencode.Dockerfile
        uses: hadolint/hadolint-action@v3.1.0
        with:
          dockerfile: base/qwencode.Dockerfile
          config: .github/lint/.hadolint.yaml
          verbose: true

      - name: Run hadolint on golang/Dockerfile
        uses: hadolint/hadolint-action@v3.1.0
        with:
          dockerfile: golang/Dockerfile
          config: .github/lint/.hadolint.yaml
          verbose: true

      - name: Run hadolint on ansible/Dockerfile
        uses: hadolint/hadolint-action@v3.1.0
        with:
          dockerfile: ansible/Dockerfile
          config: .github/lint/.hadolint.yaml
          verbose: true

      - name: Upload SARIF to Security tab
        uses: github/codeql-action/upload-sarif@v3
        if: always()
        with:
          sarif_file: hadolint.sarif
          category: hadolint

      - name: Run actionlint
        uses: rhysd/actionlint@v1
        with:
          config-file: .github/lint/.actionlint.yaml

      - name: Post actionlint results as PR comments
        uses: reviewdog/action-actionlint@v1
        with:
          github_token: ${{ secrets.github_token }}
          reporter: github-pr-review
          level: error
```

      - name: Set up QEMU
        uses: docker/setup-qemu-action@v4

      - name: Extract metadata
        id: meta
        uses: docker/metadata-action@v6
        with:
          images: ghcr.io/red55/sandbox-${{ matrix.variant }}
          tags: |
            type=pr,prefix=
            type=sha,prefix=
            type=ref,event=branch

      - name: Build image
        uses: docker/build-push-action@v7
        with:
          context: .
          file: Makefile
          tags: ${{ steps.meta.outputs.tags }}
          labels: ${{ steps.meta.outputs.labels }}
          cache-from: type=gha
          cache-to: type=gha,mode=max
          load: true
          # Note: We use make all to build, not docker/build-push-action directly
          # The build step below runs make to build all variants

      - name: Build all variants with make
        run: |
          make pull
          make all
        env:
          GHCR_TOKEN: ${{ secrets.GITHUB_TOKEN }}
```
