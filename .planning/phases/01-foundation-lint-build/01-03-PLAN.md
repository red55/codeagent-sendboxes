```yaml
name: Release Pipeline

on:
  push:
    branches: [main]
    tags: ['v*']
  workflow_dispatch:

permissions:
  contents: read
  packages: write
  security-events: write

concurrency:
  group: ${{ github.workflow }}-${{ github.ref }}
  cancel-in-progress: true

jobs:
  lint:
    name: Lint
    runs-on: ubuntu-latest
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Run hadolint on all Dockerfiles
        id: hadolint
        uses: hadolint/hadolint-action@v3.1.0
        with:
          dockerfile: base/Dockerfile
          config: .github/lint/.hadolint.yaml
          verbose: true

      - name: Run hadolint on base/opencode.Dockerfile
        uses: hadolint/hadolint-action@v3.1.0
        with:
          dockerfile: base/opencode.Dockerfile
          config: .github/lint/.hadolint.yaml
          verbose: true

      - name: Run hadolint on base/qwencode.Dockerfile
        uses: hadolint/hadolint-action@v3.1.0
        with:
          dockerfile: base/qwencode.Dockerfile
          config: .github/lint/.hadolint.yaml
          verbose: true

      - name: Run hadolint on golang/Dockerfile
        uses: hadolint/hadolint-action@v3.1.0
        with:
          dockerfile: golang/Dockerfile
          config: .github/lint/.hadolint.yaml
          verbose: true

      - name: Run hadolint on ansible/Dockerfile
        uses: hadolint/hadolint-action@v3.1.0
        with:
          dockerfile: ansible/Dockerfile
          config: .github/lint/.hadolint.yaml
          verbose: true

      - name: Upload SARIF to Security tab
        uses: github/codeql-action/upload-sarif@v3
        if: always()
        with:
          sarif_file: hadolint.sarif
          category: hadolint

      - name: Run actionlint
        uses: rhysd/actionlint@v1
        with:
          config-file: .github/lint/.actionlint.yaml

  build:
    name: Build (${{ matrix.variant }})
    needs: lint
    runs-on: ubuntu-latest
    if: github.event_name != 'pull_request'
    strategy:
      fail-fast: false
      matrix:
        variant: [oc-go, qc-go, oc-ansible, qc-ansible]
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Set up Docker Buildx
        uses: docker/setup-buildx-action@v4

      - name: Set up QEMU
        uses: docker/setup-qemu-action@v4

      - name: Login to GHCR
        uses: docker/login-action@v4
        with:
          registry: ghcr.io
          username: ${{ github.actor }}
          password: ${{ secrets.GITHUB_TOKEN }}

      - name: Extract metadata
        id: meta
        uses: docker/metadata-action@v6
        with:
          images: ghcr.io/red55/sandbox-${{ matrix.variant }}
          tags: |
            type=sha,prefix=
            type=ref,event=branch
            type=ref,event=tag
            type=raw,value=latest,enable={{is_default_branch}}

      - name: Build and push all variants with make
        run: |
          make pull
          make all
        env:
          VER: ${{ github.sha }}

      - name: Build and push individual variants
        uses: docker/build-push-action@v7
        with:
          context: .
          file: Makefile
          push: true
          tags: ${{ steps.meta.outputs.tags }}
          labels: ${{ steps.meta.outputs.labels }}
          cache-from: type=gha
          cache-to: type=gha,mode=max
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| CI Runner → ghcr.io | Pushing images requires packages:write permission |
| CI Runner → Docker Hub | Pulling base images from untrusted registries |
| Tag push → Image publish | Untrusted tags could publish malicious images |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01 | Spoofing | Tag-based pushes | mitigate | Only v* tags trigger release pipeline; tag creation requires repo maintainer |
| T-03-02 | Tampering | Workflow file | mitigate | Source-controlled, reviewed via PR |
| T-03-03 | Information Disclosure | Build logs | accept | Logs visible to repo collaborators; no secrets in logs |
| T-03-04 | Elevation of Privilege | GITHUB_TOKEN | mitigate | Token scoped to packages:write only; cannot access other secrets |
</threat_model>
